## 最短路问题
### 单源最短路
#### 正权图dijkstra
```cpp
// const ll INFLL = 1e18;
// using ll = long long;
// vector<Edge> adj[N];
vector<ll> dis(n + 1, INFLL);
auto dijkstra = [&](int s = 1) -> void {
    dis.assign(n + 1, INFLL);
    using pli = pair<ll, int>;
    priority_queue<pli, vector<pli>, greater<pli>> q;
    q.emplace(0, s);
    dis[s] = 0;
    while (!q.empty()) {
        auto [dx, x] = q.top();
        q.pop();
        if (dx > dis[x]) continue;
        for (const auto& [y, dy] : adj[x]) {
            if (dis[x] + dy < dis[y]) {
                dis[y] = dis[x] + dy;
                q.emplace(dis[y], y);
            }
        }
    }
};
```
